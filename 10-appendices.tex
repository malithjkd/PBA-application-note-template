\section{Data File Format Specification}
\label{app:data}

%%
%% WRITING GUIDE — DATA FORMAT:
%%   Documenting the CSV format makes the data FAIR-Interoperable:
%%   any tool can parse it correctly without asking the author.
%%

The force logger outputs a UTF-8 encoded CSV file with the following
column structure:

\begin{configcode}[force\_log.csv — Header and example rows]
\begin{lstlisting}
timestamp_ms,time_s,ain_voltage_V,force_N,force_setpoint_N,axis_state
0,0.000,-0.002,-0.200,5.000,ENABLED
10,0.010,-0.001,-0.100,5.000,ENABLED
20,0.020,0.051,5.100,5.000,ENABLED
\end{lstlisting}
\end{configcode}

\begin{center}
\begin{tabularx}{\textwidth}{@{} >{\ttfamily}l l X @{}}
  \toprule
  \textbf{Column} & \textbf{Type} & \textbf{Description} \\
  \midrule
  timestamp\_ms & integer & Elapsed time since logging start [\si{\milli\second}] \\
  \rowcolor{PBALightGray}
  time\_s & float & Elapsed time [\si{\second}] \\
  ain\_voltage\_V & float & Raw AIN~1 voltage [\si{\volt}], offset not subtracted \\
  \rowcolor{PBALightGray}
  force\_N & float & Calibrated force: \code{(ain - offset) * scale} [\si{\newton}] \\
  force\_setpoint\_N & float & Commanded force setpoint at time of sample [\si{\newton}] \\
  \rowcolor{PBALightGray}
  axis\_state & string & ACS axis state string from \code{MFLAGS} \\
  \bottomrule
\end{tabularx}
\end{center}

\section{Python Environment Setup}
\label{app:python-env}

Install the required Python libraries:

\begin{pythoncode}[Terminal — install dependencies]
\begin{lstlisting}
# Recommended: use a virtual environment
python -m venv .venv
.venv\Scripts\activate      # Windows
# source .venv/bin/activate   # Linux / macOS

pip install acspy numpy matplotlib pandas
\end{lstlisting}
\end{pythoncode}

\begin{notebox}
  The \code{acspy} library is the official Python API for ACS SPiiPlus
  controllers.  Download from the ACS Motion Control customer portal
  or request from PBA Systems support.  It requires the ACS C\# runtime
  which is included with the SPiiPlus MMI installation.
\end{notebox}
